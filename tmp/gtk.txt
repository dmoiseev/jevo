function markup(label::Gtk.GtkLabel,str)
	ccall((:gtk_label_set_markup,Gtk.libgtk),Void,(Ptr{Gtk.GObject},Ptr{UInt8}),label,str)
	return label
end
	
w = Gtk.@Window("test")
l = Gtk.@Label("Label")
push!(w, l)
showall(w)

destroy(w)
